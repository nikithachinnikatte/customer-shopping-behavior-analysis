#create column purchase frequency days
#create a dictionary first
# clean column
df['frequency_of_purchases'] = df['frequency_of_purchases'].str.strip().str.lower()
df = df.dropna(subset=['purchase_frequency_days','frequency_of_purchases'], how='all')

# mapping dictionary
frequency_mapping = {
    'fortnight':14,
    'week':7,
    'weekly':7,
    'monthly':30,
    'quarterly':90,
    'biweekly':14,
    'annually':365,
    'every 3 months':90
}

# create new column
df['purchase_frequency_days'] = df['frequency_of_purchases'].map(frequency_mapping)
